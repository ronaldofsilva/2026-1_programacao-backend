app.put("/people/:id", (req, res) => {
    const index = buscaPessoa(req.params.id);
    const {name, age} = req.body;    
     if (index === -1) {

    people[index].name = name;

app.get("/people/busca", (req, res) => {
    const namePerson = req.query.nome;
    try {
      const personByName = buscaPessoaPorNome(namePerson);
      res.status(200).json(personByName);
    } catch (erro) {
      res.status(500).json({ message: `${erro.message} - falha na busca` });
    }
});


function buscaPessoa(id){
    return people.findIndex(person => person.id === Number(id));

function buscaPessoaPorNome(name) {
    return people.find(person => person.name === name);
